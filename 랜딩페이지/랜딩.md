  <!-- ds-kw-js (신규 애니메이션 로직) -->
  <script id="ds-kw-js">
  (function(){
    const cv   = document.getElementById('ds-kw-canvas');
    const ctx  = cv.getContext('2d');
    const sec  = document.getElementById('vanta-birds-bg');
    const sBtn = document.getElementById('ds-search-btn');

    function setSize(){
      cv.width  = sec.offsetWidth;
      cv.height = sec.offsetHeight;
    }
    setSize();
    window.addEventListener('resize', setSize);

    const WORDS = ['합판','각재','석고보드','단열재','MDF','OSB','창호','집성목','코팅합판','방음재','플로링','흡음재','문틀','마루','방화문','목재','보드','텍스','단열','자재'];

    // sCenter Fix: Revised Logic to capture precise coordinates
    function sCenter() {
      const sbox = document.querySelector('.search_btn') || document.querySelector('fieldset');
      const hr = document.getElementById('vanta-birds-bg').getBoundingClientRect();
      const sr = sbox.getBoundingClientRect();
      return {
        x: sr.left - hr.left + sr.width  * 0.5,
        y: sr.top  - hr.top  + sr.height * 0.5
      };
    }

    function mkP(initial){
      const w=cv.width, h=cv.height;
      let x,y;
      if(initial){ x=20+Math.random()*(w-40); y=20+Math.random()*(h-40); }
      else {
        const side=Math.floor(Math.random()*4);
        if(side===0){x=Math.random()*w;y=-30;}
        else if(side===1){x=w+30;y=Math.random()*h;}
        else if(side===2){x=Math.random()*w;y=h+30;}
        else{x=-30;y=Math.random()*h;}
      }

      const isGold = Math.random() < 0.3;
      const alpha = 0.055 + Math.random() * 0.065;
      return {
        word: WORDS[Math.floor(Math.random()*WORDS.length)],
        x, y,
        vx: (Math.random()-.5)*.4,
        vy: (Math.random()-.5)*.4,
        size: 11 + Math.random()*7,
        alpha,
        maxA: alpha,
        phase: Math.random()*Math.PI*2,
        amp: 7 + Math.random()*10,
        dur: 14000 + Math.random()*10000,
        sucking: false,
        clicked: false,
        isGold,
        _oy: 0
      };
    }

    let pts = Array.from({length:28},()=>mkP(true));
    const T0 = performance.now();
    let globalSucking = false;

    if(sBtn) {
      sBtn.addEventListener('click', () => {
        globalSucking = true;
        pts.forEach(p => { p.clicked = true; p.sucking = true; });
      });
    }

    function frame(ts){
      const elapsed = (ts-T0);
      ctx.clearRect(0,0,cv.width,cv.height);
      const sc = sCenter();
      const autoSuckOn = elapsed > 1600;

      pts.forEach((p,i)=>{
        if(autoSuckOn && !p.sucking && Math.random() < 0.004) p.sucking = true;

        if(p.sucking){
          const dx = sc.x - p.x;
          const dy = sc.y - p.y;
          const dist = Math.hypot(dx,dy);
          
          if(p.clicked) {
            p.vx += dx * 0.06;
            p.vy += dy * 0.06;
            p.vx *= 0.82;
            p.vy *= 0.82;
          } else {
            p.vx += dx * 0.014;
            p.vy += dy * 0.014;
            p.vx *= 0.86;
            p.vy *= 0.86;
          }

          p.alpha = Math.max(0, p.alpha - 0.008);
          p.x += p.vx; p.y += p.vy;

          if(dist < 16 || p.alpha <= 0){ pts[i] = mkP(false); return; }
        } else {
          const oy = Math.sin(ts/p.dur*Math.PI*2+p.phase)*p.amp;
          p.y += (oy-p._oy)*0.28; p._oy = oy;
          p.x += p.vx;
          p.alpha = Math.min(p.maxA, p.alpha + 0.003);
          if(p.x < -60 || p.x > cv.width+60 || p.y < -60 || p.y > cv.height+60){ pts[i] = mkP(false); return; }
        }
        
        ctx.save();
        ctx.font = `900 ${p.size}px Pretendard,sans-serif`;
        ctx.fillStyle = p.isGold || p.sucking ? `rgba(201,168,76,${Math.max(0,p.alpha)})` : `rgba(18,54,40,${Math.max(0,p.alpha)})`;
        ctx.fillText(p.word, p.x, p.y);
        ctx.restore();
      });

      if(autoSuckOn || globalSucking){
        const alphaGlow = Math.min(0.25, (elapsed-1600)*0.0001 + 0.13);
        const g = ctx.createRadialGradient(sc.x, sc.y, 0, sc.x, sc.y, 100);
        g.addColorStop(0, `rgba(201,168,76,${alphaGlow})`);
        g.addColorStop(1, 'rgba(201,168,76,0)');
        ctx.fillStyle = g;
        ctx.beginPath(); ctx.arc(sc.x, sc.y, 100, 0, Math.PI*2); ctx.fill();
      }

      requestAnimationFrame(frame);
    }
    requestAnimationFrame(frame);
  })();
  </script>

## 배경 애니메이션 추가 (신규)
`<section>` 안에 `<canvas id="ds-kw-canvas">` 추가 후 아래 동작 구현:

### 동작 설명
1. 흰 배경 위에 건축 키워드들(`합판, 각재, 석고보드, 단열재, MDF, OSB, 창호, 집성목, 코팅합판, 방음재, 플로링, 흡음재, 문틀, 마루, 방화문, 목재, 보드, 텍스`)이 매우 흐리게(opacity 0.05~0.12) 배경에서 둥둥 부유
2. 페이지 로드 약 1.6초 후 키워드들이 하나씩 순서대로 검색창 방향으로 빨려들어감
3. 검색창 도달 시 사라지고 새 키워드가 화면 밖에서 등장 → 무한 반복
4. 흡입 시작 후 검색창 주변에 골드(`rgba(201,168,76,0.13)`) 글로우 은은하게 활성화

### 캔버스 CSS
```css
#ds-kw-canvas {
  position: absolute;
  inset: 0;
  width: 100%;
  height: 100%;
  pointer-events: none;
  z-index: 0;
}
```

### 키워드 텍스트 스타일
```
font: 900 11~18px Pretendard, sans-serif
fillStyle: rgba(18, 54, 40, 0.05~0.12)
```

### 흡입 로직
- 각 키워드는 매 프레임 0.4% 확률로 흡입 모드 전환
- 흡입 시 검색창 중심 좌표로 가속 (dx * 0.014, dy * 0.014)
- 감속계수 0.86 적용
- 검색창 16px 이내 도달 또는 alpha <= 0 이면 재생성

## 최종 섹션 구조
```html
<section id="vanta-birds-bg" class="main_section main_visual_section">

  <!-- 키워드 부유 캔버스 (신규) -->
  <canvas id="ds-kw-canvas"></canvas>

  <div class="main_visual_area">
    <div class="section_title">
      <div class="tit">건축자재의 쿠팡</div>
      <div class="eng custom_font">대산</div>
      <div class="txt">건축자재도 이제 클릭 한 번이면 됩니다</div>
    </div>

    <!-- 검색창 모듈 (기존 유지) -->
    <div module="Layout_SearchHeader">
      <!--@css(/css/module/layout/searchHeader.css)-->
      <!--
      $product_page=/product/detail.html
      $category_page=/product/list.html
      -->
      <fieldset>
        <legend>검색</legend>
        {$form.keyword}<a href="javascript:void(0)" onclick="{$action_search_submit}" class="search_btn">
          <img src="//ecimg.cafe24img.com/pg2383b21973322017/daesan3833/web/upload/free_design/search_btn_icon.png" alt="검색하기">
        </a>
        <ul class="autoDrop" id="{$search_auto_id}"></ul>
      </fieldset>
    </div>

    <!-- main_cate (절대 수정 금지) -->
    <ul class="main_cate" df-banner-code="main-cate" hidden>
      <li df-banner-clone><a href="{#href}" target="{#target}"><p>{#item}</p><div class="contents">{#title}</div></a></li>
    </ul>
  </div>

  <!-- 캔버스 JS (신규) -->
  <script>
  (function(){
    const cv   = document.getElementById('ds-kw-canvas');
    const ctx  = cv.getContext('2d');
    const sec  = document.getElementById('vanta-birds-bg');
    const sbox = document.querySelector('.search_btn') || document.querySelector('fieldset');

    function setSize(){
      cv.width  = sec.offsetWidth;
      cv.height = sec.offsetHeight;
    }
    setSize();
    window.addEventListener('resize', setSize);

    const WORDS = ['합판','각재','석고보드','단열재','MDF','OSB','창호','집성목','코팅합판','방음재','플로링','흡음재','문틀','마루','방화문','목재','보드','텍스','단열','자재'];

    function sCenter(){
      const sr = sbox.getBoundingClientRect();
      const hr = sec.getBoundingClientRect();
      return { x: sr.left-hr.left+sr.width*.5, y: sr.top-hr.top+sr.height*.5 };
    }

    function mkP(initial){
      const w=cv.width, h=cv.height;
      let x,y;
      if(initial){ x=20+Math.random()*(w-40); y=20+Math.random()*(h-40); }
      else {
        const side=Math.floor(Math.random()*4);
        if(side===0){x=Math.random()*w;y=-24;}
        else if(side===1){x=w+24;y=Math.random()*h;}
        else if(side===2){x=Math.random()*w;y=h+24;}
        else{x=-24;y=Math.random()*h;}
      }
      return {
        word:WORDS[Math.floor(Math.random()*WORDS.length)],
        x,y,
        vx:(Math.random()-.5)*.4,
        vy:(Math.random()-.5)*.4,
        size:11+Math.random()*7,
        alpha:initial?Math.random()*0.08:0,
        maxA:0.055+Math.random()*0.065,
        phase:Math.random()*Math.PI*2,
        amp:7+Math.random()*10,
        dur:14+Math.random()*10,
        sucking:false,
        _oy:0
      };
    }

    let pts=Array.from({length:24},()=>mkP(true));
    const T0=performance.now();
    let t=0;

    function frame(ts){
      const elapsed=(ts-T0)/1000;
      ctx.clearRect(0,0,cv.width,cv.height);
      const sc=sCenter();
      const suckOn=elapsed>1.6;

      pts.forEach((p,i)=>{
        if(suckOn&&!p.sucking&&Math.random()<0.004) p.sucking=true;
        if(p.sucking){
          const dx=sc.x-p.x,dy=sc.y-p.y;
          const dist=Math.hypot(dx,dy);
          p.vx+=dx*0.014; p.vy+=dy*0.014;
          p.vx*=0.86; p.vy*=0.86;
          p.alpha=Math.max(0,p.alpha-0.005);
          p.x+=p.vx; p.y+=p.vy;
          if(dist<16||p.alpha<=0){pts[i]=mkP(false);return;}
        } else {
          const oy=Math.sin(t/p.dur*Math.PI*2+p.phase)*p.amp;
          p.y+=(oy-p._oy)*0.28; p._oy=oy;
          p.x+=p.vx;
          p.alpha=Math.min(p.maxA,p.alpha+0.003);
          if(p.x<-40||p.x>cv.width+40||p.y<-40||p.y>cv.height+40){pts[i]=mkP(false);return;}
        }
        ctx.save();
        ctx.font=`900 ${p.size}px Pretendard,sans-serif`;
        ctx.fillStyle=`rgba(18,54,40,${Math.max(0,p.alpha)})`;
        ctx.fillText(p.word,p.x,p.y);
        ctx.restore();
      });

      if(suckOn){
        const a=Math.min(0.13,(elapsed-1.6)*0.05);
        const g=ctx.createRadialGradient(sc.x,sc.y,0,sc.x,sc.y,90);
        g.addColorStop(0,`rgba(201,168,76,${a})`);
        g.addColorStop(1,'rgba(201,168,76,0)');
        ctx.fillStyle=g;
        ctx.beginPath();ctx.arc(sc.x,sc.y,90,0,Math.PI*2);ctx.fill();
      }

      t+=0.016;
      requestAnimationFrame(frame);
    }
    requestAnimationFrame(frame);
  })();
  </script>

</section>
```
